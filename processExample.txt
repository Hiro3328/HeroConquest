must have:
	*_ Current Save Slot
	*_ path/file to the save slot
	*_ the vars to update

reads the information on the file
search for the older value and replace it with the new one:
	*_ search for the line with the nameOfTheVar
	*_ delete the line
	*_ write(f"{nameOfTheVar} = {varValue}") to the line



vars to update must be a dictionary:
	*_ separate the currentSave. from the var name
	*_ add the varName, varValue to the dictionary


def updateSaves(fileName, changedVars) -> None:
    path = f'saves/{fileName}.py'
    with open(path, "r+") as file:
        contents = file.read()
        lines = contents.split("\n")
        for i, line in enumerate(lines):
            for varName, varValue in changedVars.items():
                if varName in line:
                    if type(varValue) == str:
                        lines[i] = f"{varName} = '{varValue}'"
                    else:
                        lines[i] = f"{varName} = {varValue}"
                    break
        file.seek(0)
        file.write("\n".join(lines))
        file.close()
